-- Services
local Players = game:GetService("Players")

-- Globals
_G.ESPToggle = false
_G.FlyEnabled = false
_G.FlySpeed = 70
_G.ESPTransparency = 0.5
local flyConn = nil
local velocity = nil

-- ESP Table
local colourTable = {
    Red = Color3.fromRGB(255, 0, 0),
}
local colourNames = {}
for name in pairs(colourTable) do table.insert(colourNames, name) end
local selectedColour = "Red"

-- ESP Logic
local function getCharacter(player)
    return player.Character or workspace:FindFirstChild(player.Name)
end

local function addHighlight(player, character)
    if player == LocalPlayer then return end
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if hrp and not hrp:FindFirstChild("Highlight") then
        local hl = Instance.new("Highlight")
        hl.Name = "Highlight"
        hl.Adornee = character
        hl.Parent = hrp
        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        hl.FillColor = colourTable[selectedColour]
        hl.OutlineColor = Color3.new(1, 1, 1)
        hl.FillTransparency = _G.ESPTransparency
    elseif hrp and hrp:FindFirstChild("Highlight") then
        local hl = hrp:FindFirstChild("Highlight")
        hl.FillColor = colourTable[selectedColour]
        hl.FillTransparency = _G.ESPTransparency
    end
end

local function removeHighlight(character)
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if hrp then
        local hl = hrp:FindFirstChild("Highlight")
        if hl then hl:Destroy() end
    end
end

local function updateESP()
    for _, player in pairs(Players:GetPlayers()) do
        local char = getCharacter(player)
        if char then
            if _G.ESPToggle then
                addHighlight(player, char)
            else
                removeHighlight(char)
            end
        end
    end
end

RunService.RenderStepped:Connect(updateESP)
Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(char)
        if _G.ESPToggle then
            addHighlight(player, char)
        end
    end)
end)
Players.PlayerRemoving:Connect(function(p)
    removeHighlight(p.Character)
end)

-- FLY Logic
local function toggleFly(enabled)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    local hrp = char.HumanoidRootPart
    if flyConn then flyConn:Disconnect() end
    if velocity then velocity:Destroy() end
    if enabled then
        velocity = Instance.new("BodyVelocity")
        velocity.MaxForce = Vector3.new(1e6, 1e6, 1e6)
        velocity.Velocity = Vector3.zero
        velocity.Parent = hrp
        flyConn = RunService.RenderStepped:Connect(function()
            local camCF = workspace.CurrentCamera.CFrame
            local move = Vector3.zero
            if UIS:IsKeyDown(Enum.KeyCode.W) then move += camCF.LookVector end
            if UIS:IsKeyDown(Enum.KeyCode.S) then move -= camCF.LookVector end
            if UIS:IsKeyDown(Enum.KeyCode.A) then move -= camCF.RightVector end
            if UIS:IsKeyDown(Enum.KeyCode.D) then move += camCF.RightVector end
            if UIS:IsKeyDown(Enum.KeyCode.Space) then move += Vector3.new(0, 1, 0) end
            if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then move -= Vector3.new(0, 1, 0) end
            velocity.Velocity = move.Magnitude > 0 and move.Unit * _G.FlySpeed or Vector3.zero
        end)
    end
end

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(1)
    if _G.FlyEnabled then
        toggleFly(true)
    end
end)
